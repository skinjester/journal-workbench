FX = postprocess.FX()

because you'll never do this:
FX1 = postprocess.FX()

rgb = postprocess.FX.median_blur(rgb, **fx['params'])

from postprocess.FX import median_blur
rgb = median_blur(rgb, **fx['params'])

rgb = postprocess.median_blur(rgb, **fx['params'])
